dotnet tool install -g dotnet-reportgenerator-globaltool

reportgenerator -reports:".\TestResults\ae3b3549-538c-474a-b364-da8a8bdd924d(اسم الملف)\coverage.cobertura.xml" -targetdir:"CodeTesting(اسم لل folder)" -reporttypes:HTML
